# Format-on-Edit Hook — Language-Aware Generation

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Replace the static `templates/hooks/format-on-edit.mjs` with a generated hook that only includes formatter blocks for languages actually detected in the target project.

**Architecture:** Add `lib/generate/format-hook.mjs` (pure generator, no IO) that accepts a `ProjectProfile` and returns hook source as a string. `lib/plan.mjs` calls it and adds the artifact with `content:` (generated inline) instead of `copyFrom:` (static copy). Follow the identical pattern used by `lib/generate/verify.mjs` + `renderVerifyOnStop`.

**Tech Stack:** Node ≥18, ESM `.mjs`, JSDoc types, `node:child_process` execFileSync in generated hook output.

## Global Constraints

- All source files: `.mjs` ESM with JSDoc — no TypeScript, no build step
- `lib/` is pure and side-effect-free — no filesystem IO in generators
- Generated hook must never block an edit: all formatter failures exit 0
- Generated JS must not use template literals / backticks internally (use string concatenation) so embedding in this module's template literal needs no escaping — same rule as `verify.mjs`
- Tests: `node:test` + `node:assert/strict`, matching style in `tests/verify-hook.test.mjs`
- `npm test` must pass after every task

```js
/**
 * Generate a language-aware format-on-edit PostToolUse hook.
 * Returns JS source as a string. Pure: no IO.
 *
 * The generated code uses string concatenation, not template literals,
 * so embedding it inside this module's template literal needs no escaping.
 *
 * @module generate/format-hook
 */

/** @typedef {import('../profile.mjs').ProjectProfile} ProjectProfile */

/**
 * @param {ProjectProfile} profile
 * @returns {string}
 */
export function renderFormatOnEdit(profile) {
  const names = new Set((profile.languages || []).map((l) => l.name));

  const hasJsLike = names.has("JavaScript") || names.has("TypeScript") ||
    names.has("Vue") || names.has("Svelte") || names.has("Astro");
  const hasPHP   = names.has("PHP");
  const hasPython = names.has("Python");
  const hasGo    = names.has("Go");
  const hasRust  = names.has("Rust");
  const hasRuby  = names.has("Ruby");
  const hasDart  = names.has("Dart");
  const hasCC    = names.has("C") || names.has("C++");

  // No languages detected → include JS/TS as safe default.
  const forceJsLike = !hasJsLike && !hasPHP && !hasPython && !hasGo &&
    !hasRust && !hasRuby && !hasDart && !hasCC;

  /** @type {{ condition: string, body: string }[]} */
  const blocks = [];

  if (hasJsLike || forceJsLike) blocks.push(JS_BLOCK);
  if (hasPHP)    blocks.push(PHP_BLOCK);
  if (hasPython) blocks.push(PYTHON_BLOCK);
  if (hasGo)     blocks.push(GO_BLOCK);
  if (hasRust)   blocks.push(RUST_BLOCK);
  if (hasRuby)   blocks.push(RUBY_BLOCK);
  if (hasDart)   blocks.push(DART_BLOCK);
  if (hasCC)     blocks.push(CC_BLOCK);

  return buildHook(blocks);
}

/**
 * @param {{ condition: string, body: string }[]} blocks
 * @returns {string}
 */
function buildHook(blocks) {
  const chain = blocks
    .map((b, i) => (i === 0 ? "if" : "else if") + " (" + b.condition + ") {\n" + b.body + "\n  }")
    .join(" ");

  return "#!/usr/bin/env node\n" +
    "/**\n" +
    " * PostToolUse hook: format the file Claude just edited.\n" +
    " * Generated by aia-harness — language-aware. Never blocks.\n" +
    " */\n" +
    "import fs from \"node:fs\";\n" +
    "import path from \"node:path\";\n" +
    "import { execFileSync } from \"node:child_process\";\n" +
    "\n" +
    "function readStdin() {\n" +
    "  try { return fs.readFileSync(0, \"utf8\"); } catch { return \"\"; }\n" +
    "}\n" +
    "\n" +
    "let event = {};\n" +
    "try { event = JSON.parse(readStdin() || \"{}\"); } catch { process.exit(0); }\n" +
    "\n" +
    "const file = event && event.tool_input && (event.tool_input.file_path || event.tool_input.path);\n" +
    "if (!file || typeof file !== \"string\") process.exit(0);\n" +
    "\n" +
    "const projectDir = process.env.CLAUDE_PROJECT_DIR || process.cwd();\n" +
    "const ext = path.extname(file).toLowerCase();\n" +
    "\n" +
    "function localBin(name) {\n" +
    "  var p = path.join(projectDir, \"node_modules\", \".bin\", name);\n" +
    "  return fs.existsSync(p) ? p : null;\n" +
    "}\n" +
    "\n" +
    "try {\n" +
    "  " + chain + "\n" +
    "} catch {\n" +
    "  // Formatting is best-effort; never block the edit.\n" +
    "}\n" +
    "\n" +
    "process.exit(0);\n";
}

/** @type {{ condition: string, body: string }} */
const JS_BLOCK = {
  condition: "[\".js\",\".jsx\",\".ts\",\".tsx\",\".mjs\",\".cjs\",\".mts\",\".cts\"," +
    "\".json\",\".jsonc\",\".css\",\".scss\",\".sass\",\".less\"," +
    "\".md\",\".mdx\",\".html\",\".htm\",\".yaml\",\".yml\"," +
    "\".vue\",\".svelte\",\".astro\"].includes(ext)",
  body: "    var biome = localBin(\"biome\");\n" +
    "    var prettier = localBin(\"prettier\");\n" +
    "    if (biome) execFileSync(biome, [\"format\", \"--write\", file], { stdio: \"ignore\" });\n" +
    "    else if (prettier) execFileSync(prettier, [\"--write\", file], { stdio: \"ignore\" });",
};

/** @type {{ condition: string, body: string }} */
const PHP_BLOCK = {
  condition: "[\".php\",\".phtml\"].includes(ext)",
  body: "    var pint = path.join(projectDir, \"vendor\", \"bin\", \"pint\");\n" +
    "    if (fs.existsSync(pint)) execFileSync(pint, [file], { stdio: \"ignore\" });",
};

/** @type {{ condition: string, body: string }} */
const PYTHON_BLOCK = {
  condition: "[\".py\",\".pyi\"].includes(ext)",
  body: "    var venvBin = path.join(projectDir, \".venv\", \"bin\");\n" +
    "    var ruff = fs.existsSync(path.join(venvBin, \"ruff\")) ? path.join(venvBin, \"ruff\") : \"ruff\";\n" +
    "    var black = fs.existsSync(path.join(venvBin, \"black\")) ? path.join(venvBin, \"black\") : \"black\";\n" +
    "    try { execFileSync(ruff, [\"format\", \"--quiet\", file], { stdio: \"ignore\" }); }\n" +
    "    catch { try { execFileSync(black, [\"--quiet\", file], { stdio: \"ignore\" }); } catch { /* none */ } }",
};

/** @type {{ condition: string, body: string }} */
const GO_BLOCK = {
  condition: "ext === \".go\"",
  body: "    execFileSync(\"gofmt\", [\"-w\", file], { stdio: \"ignore\" });",
};

/** @type {{ condition: string, body: string }} */
const RUST_BLOCK = {
  condition: "ext === \".rs\"",
  body: "    execFileSync(\"rustfmt\", [file], { stdio: \"ignore\" });",
};

/** @type {{ condition: string, body: string }} */
const RUBY_BLOCK = {
  condition: "[\".rb\",\".erb\"].includes(ext)",
  body: "    try {\n" +
    "      execFileSync(\"bundle\", [\"exec\", \"rubocop\", \"--autocorrect-all\", \"--no-server\", file],\n" +
    "        { cwd: projectDir, stdio: \"ignore\" });\n" +
    "    } catch {\n" +
    "      execFileSync(\"rubocop\", [\"--autocorrect-all\", \"--no-server\", file], { stdio: \"ignore\" });\n" +
    "    }",
};

/** @type {{ condition: string, body: string }} */
const DART_BLOCK = {
  condition: "ext === \".dart\"",
  body: "    execFileSync(\"dart\", [\"format\", file], { stdio: \"ignore\" });",
};

/** @type {{ condition: string, body: string }} */
const CC_BLOCK = {
  condition: "[\".c\",\".h\",\".cc\",\".cpp\",\".cxx\",\".hpp\"].includes(ext)",
  body: "    var hasConfig = [\".clang-format\",\"_clang-format\"].some(\n" +
    "      function(f) { return fs.existsSync(path.join(projectDir, f)); });\n" +
    "    if (hasConfig) execFileSync(\"clang-format\", [\"-i\", file], { stdio: \"ignore\" });",
};
```

```js
import { test } from "node:test";
import assert from "node:assert/strict";
import fs from "node:fs";
import os from "node:os";
import path from "node:path";
import { execFileSync } from "node:child_process";
import { renderFormatOnEdit } from "../lib/generate/format-hook.mjs";

/** Minimal profile stub with given language names. */
function profile(langNames) {
  return {
    languages: langNames.map((name) => ({ name, type: "programming", bytes: 1000, files: 10, share: 1 })),
  };
}

/** Write generated hook to a temp file; return its path. */
function genHook(langNames) {
  const src = renderFormatOnEdit(/** @type {any} */ (profile(langNames)));
  const dir = fs.mkdtempSync(path.join(os.tmpdir(), "aia-fmt-"));
  const file = path.join(dir, "format-on-edit.mjs");
  fs.writeFileSync(file, src);
  return file;
}

/** Run the hook with a given stdin event; return exit code. */
function run(hookFile, projectDir, event) {
  try {
    execFileSync(process.execPath, [hookFile], {
      input: JSON.stringify(event),
      env: { ...process.env, CLAUDE_PROJECT_DIR: projectDir },
      encoding: "utf8",
    });
    return 0;
  } catch (err) {
    return /** @type {any} */ (err).status ?? 1;
  }
}

test("JS/TS project: includes biome/prettier block, excludes python/go", () => {
  const src = renderFormatOnEdit(/** @type {any} */ (profile(["TypeScript"])));
  assert.match(src, /biome/);
  assert.match(src, /prettier/);
  assert.ok(!src.includes("ruff"));
  assert.ok(!src.includes("gofmt"));
});

test("Python project: includes ruff/black block, excludes biome", () => {
  const src = renderFormatOnEdit(/** @type {any} */ (profile(["Python"])));

});

test("Go project: includes gofmt block, excludes php/python", () => {
  const src = renderFormatOnEdit(/** @type {any} */ (profile(["Go"])));
  assert.match(src, /gofmt/);
  assert.ok(!src.includes("pint"));
  assert.ok(!src.includes("ruff"));
});

test("Rust project: includes rustfmt block", () => {
  const src = renderFormatOnEdit(/** @type {any} */ (profile(["Rust"])));

test("PHP project: includes pint block, excludes biome/prettier", () => {

test("empty languages: falls back to JS/TS block", () => {

  assert.match(src, /biome/);

test("returns a string starting with shebang", () => {

test("exits 0 for unknown file type (no matching block)", () => {
  const hook = genHook(["TypeScript"]);
  const proj = fs.mkdtempSync(path.join(os.tmpdir(), "aia-proj-"));
  const code = run(hook, proj, { tool_input: { file_path: "/tmp/image.png" } });

test("exits 0 when formatter binary is absent (fail-open)", () => {
  // Go block: gofmt absent → outer catch exits 0
  const hook = genHook(["Go"]);
  const proj = fs.mkdtempSync(path.join(os.tmpdir(), "aia-proj-"));

  const emptyBin = fs.mkdtempSync(path.join(os.tmpdir(), "aia-bin-"));
  const code = run(hook, proj, { tool_input: { file_path: "/tmp/main.go" } });

  assert.equal(code, 0);
  fs.rmSync(emptyBin, { recursive: true, force: true });

test("exits 0 when no file_path in event", () => {

  const proj = fs.mkdtempSync(path.join(os.tmpdir(), "aia-proj-"));

});

test("exits 0 with malformed stdin", () => {

      encoding: "utf8",

    assert.fail("hook should not throw: " + /** @type {any} */ (err).message);
```
